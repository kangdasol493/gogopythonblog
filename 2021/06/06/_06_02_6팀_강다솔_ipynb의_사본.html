def is_prime(num):
    #소수여부 판단하여 리턴
    #소수:True, 소수아님:False
    if num < 2:
        return False

    for i in range(2,num):
        if num % i ==0:
            return False
    
    return True

input_num = int(input('숫자:'))

    print('소수입니다.')
else:
    print('소수가 아닙니다.')

    if count == 0:     # 종료 조건을 만들어야함 count가 0이면 다시 호출하지 않고 끝냄

    while True:
        name=input('상품명:')
        if name =='':
            break
        num=int(input('개수:'))
        if name in dict_prod:
            dict_prod[name] += num
        else:
            dict_prod[name] = num

    print('-'*50)
    for key, value in dict_prod.items():
        print(f'{key} : {value}')

# 2.제품판매
def sell():
    while True:
        name=input('상품명:')

                dict_prod[name] -= num
                print('-'*50)
                print(f'{dict_prod[name]} 개 남았습니다.')
